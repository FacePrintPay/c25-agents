<script>
const API = 'http://localhost:8000/api/v1';
function show(id){document.querySelectorAll('.section').forEach(d=>d.classList.remove('active'));document.querySelectorAll('nav button').forEach(b=>b.classList.remove('active'));document.getElementById(id).classList.add('active');event.target.classList.add('active');if(id==='agents')loadAgents();if(id==='modules')loadModules();if(id==='logs')loadLogs();}
async function sendGlobalPrompt(){const prompt=document.getElementById('global-prompt').value.trim();const agent=document.getElementById('agent-select').value;if(!prompt)return showResponse('⚠️ Enter a command',true);showResponse('🔄 Processing...',false);try{const res=await fetch(`${API}/prompt`,{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({prompt,agent_name:agent})});const data=await res.json();showResponse(data.response,false,data.action_served);if(id==='logs')loadLogs();}catch(e){showResponse(`❌ Error: ${e.message}`,true);}document.getElementById('global-prompt').value='';}
async function executeCommand(){const cmd=document.getElementById('cmd-input').value.trim();const output=document.getElementById('cmd-output');if(!cmd)return;output.textContent='⏳ Executing...';try{const res=await fetch(`${API}/execute`,{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({prompt:cmd,agent_name:'Executor'})});const data=await res.json();output.textContent=data.output;showResponse(`✅ ${data.status}: ${cmd.substring(0,50)}...`,false);}catch(e){output.textContent=`❌ Error: ${e.message}`;showResponse(`❌ Execution failed: ${e.message}`,true);}}
async function loadAgents(){const grid=document.getElementById('agent-grid');try{const res=await fetch(`${API}/agents`);const agents=await res.json();grid.innerHTML=agents.map(a=>`<div class="card"><h3>${a.name}</h3><p class="meta">${a.role} • Port ${a.port}</p><p>${a.duty}</p><span class="badge">${a.status}</span></div>`).join('')||'<div class="card"><h3>No agents</h3></div>';}catch(e){grid.innerHTML='<div class="card"><h3>⚠️ Failed to load</h3></div>';}}
async function loadModules(){const grid=document.getElementById('module-grid');try{const res=await fetch(`${API}/modules`);const mods=await res.json();grid.innerHTML=mods.map(m=>`<div class="card"><h3>${m.name}</h3><p class="meta">$${m.price} • ${m.status}</p><p>${m.description}</p></div>`).join('')||'<div class="card"><h3>No modules</h3></div>';}catch(e){grid.innerHTML='<div class="card"><h3>⚠️ Failed to load</h3></div>';}}
async function loadLogs(){const list=document.getElementById('log-list');try{const res=await fetch(`${API}/logs?limit=20`);const logs=await res.json();list.innerHTML=logs.map(l=>`<div class="log-entry"><strong>[${l.agent_name}]</strong> ${l.prompt}<span class="timestamp">${new Date(l.timestamp).toLocaleTimeString()}</span><span class="action-badge action-${l.served_action}">${l.served_action}</span><br><small>${l.response.substring(0,120)}${l.response.length>120?'...':''}</small></div>`).join('')||'<div class="log-entry">No recent logs</div>';}catch(e){list.innerHTML='<div class="log-entry">⚠️ Failed to load logs</div>';}}
function showResponse(msg,isError,action){const el=document.getElementById('live-response');el.textContent=msg;el.className=isError?'error':'';if(action)el.textContent+=` [${action}]`;el.style.display='block';setTimeout(()=>{el.style.display='none'},8000);}
function clearResponse(){document.getElementById('live-response').style.display='none';document.getElementById('global-prompt').value='';}
// Auto-load on start
window.onload=()=>{loadAgents();loadModules();loadLogs();setInterval(loadLogs,30000);};
